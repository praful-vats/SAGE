# System initialization script
# SAGE Defense Terminal v3.2

echo "Initializing SAGE defense systems..."
load_modules() {
  for module in $(ls /opt/sage/modules/*.mod); do
    echo "Loading module: $(basename $module)"
    source $module
  done
}

# Start core services
start_core_services() {
  systemctl start sage-radar
  systemctl start sage-comms
  systemctl start sage-targeting
}

load_modules
start_core_services
echo "System initialization complete."
